def parentMultiple(*args):
    sel = cmds.ls(sl=True)
    if len(sel)>1:
        gp = sel.pop(0)
        for each in sel:
            cmds.parentConstraint(gp, each, mo=True)
    else:
        logger.warning("Select at least two objects to parent")
        dialogs.error_prompt('Selection Error', 'Select at least two objects to parent')
