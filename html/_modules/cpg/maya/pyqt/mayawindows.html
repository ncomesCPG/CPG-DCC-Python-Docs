[docs]	def eventFilter(self, obj, event):
		if obj is self:
			if event.type() == QEvent.Close:
				for callback in self._callbacks:
					self.unregister_event(callback)
				self.closeEvent(event)
				return True
			else:
				return super().eventFilter(obj, event)
